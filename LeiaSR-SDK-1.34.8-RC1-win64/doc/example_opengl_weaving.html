208float GetSRGB(float value)
209{
210 // If already in sRGB, no change.
211 if (g_colorSpace != eColorSpace::Default)
212 return value;
213
214 // Convert linear->sRGB.
215 if (value <= 0.0f)
216 return 0.0f;
217 else if (value >= 1.0f)
218 return 1.0f;
219 else if (value <= 0.0031308f)
220 return value * 12.92f;
221 else
222 return 1.055f * pow(value, 1.0f / 2.4f) - 0.055f;
223}

236bool GetNonPrimaryDisplayTopLeftCoordinate(int& x, int& y)
237{
238 // Get connected monitor info.
239 std::vector<MONITORINFOEX> mInfo;
240 mInfo.reserve(::GetSystemMetrics(SM_CMONITORS));
241 EnumDisplayMonitors(NULL, NULL, GetDefaultWindowStartPos_MonitorEnumProc, reinterpret_cast<LPARAM>(&mInfo));
242
243 // If we have multiple monitors, select the first non-primary one.
244 if (mInfo.size() > 1)
245 {
246 for (int i = 0; i < mInfo.size(); i++)
247 {
248 const MONITORINFOEX& mi = mInfo[i];
249
250 if (0 == (mi.dwFlags & MONITORINFOF_PRIMARY))
251 {
252 x = mi.rcMonitor.left;
253 y = mi.rcMonitor.top;
254 return true;
255 }
256 }
257 }
258
259 // Didn't find a non-primary, there is only one display connected.
260 x = 0;
261 y = 0;
262 return false;
263}

364HGLRC InitializeGL(HWND hWnd, HDC hDC)
365{
366 HGLRC context = NULL;
367
368 PFNWGLCHOOSEPIXELFORMATARBPROC wglChoosePixelFormatARBFunc = nullptr;
369 PFNWGLCREATECONTEXTATTRIBSARBPROC wglCreateContextAttribsARBFunc = nullptr;
370 {
371 // First create a context for the purpose of getting access to wglChoosePixelFormatARB / wglCreateContextAttribsARB.
372 PIXELFORMATDESCRIPTOR pfd = {};
373 pfd.nSize = sizeof(pfd);
374 pfd.nVersion = 1;
375 pfd.iPixelType = PFD_TYPE_RGBA;
376 pfd.dwFlags = PFD_SUPPORT_OPENGL | PFD_DRAW_TO_WINDOW | PFD_DOUBLEBUFFER;
377 pfd.cColorBits = 32;
378 pfd.cDepthBits = 16;
379 int pf = ChoosePixelFormat(hDC, &pfd);
380 if (pf == 0)
381 {
382 OnError("Failed to choose pixel format.");
383 return NULL;
384 }
385
386 if (!SetPixelFormat(hDC, pf, &pfd))
387 {
388 OnError("Failed to set pixel format.");
389 return NULL;
390 }
391
392 HGLRC context = wglCreateContext(hDC);
393 if (context == 0)
394 {
395 OnError("wglCreateContextfailed failed.");
396 return NULL;
397 }
398
399 if (!wglMakeCurrent(hDC, context))
400 {
401 OnError("wglMakeCurrent failed.");
402 return NULL;
403 }
404
405 wglChoosePixelFormatARBFunc = (PFNWGLCHOOSEPIXELFORMATARBPROC)wglGetProcAddress("wglChoosePixelFormatARB");
406 wglCreateContextAttribsARBFunc = (PFNWGLCREATECONTEXTATTRIBSARBPROC)wglGetProcAddress("wglCreateContextAttribsARB");
407
408 wglDeleteContext(context);
409
410 if (wglChoosePixelFormatARBFunc == nullptr || wglCreateContextAttribsARBFunc == nullptr)
411 {
412 OnError("wglChoosePixelFormatARB and/or wglCreateContextAttribsARB missing.");
413 return NULL;
414 }
415 }
416
417 // Now create the real context that we will be using.
418 const int iAttributes[] =
419 {
420 WGL_SUPPORT_OPENGL_ARB, GL_TRUE,
421 WGL_COLOR_BITS_ARB, 32,
422 WGL_DEPTH_BITS_ARB, 16,
423 WGL_DOUBLE_BUFFER_ARB, GL_TRUE,
424 WGL_FRAMEBUFFER_SRGB_CAPABLE_ARB, GL_TRUE,
425 0, 0
426 };
427
428 const float fAttributes[] = { 0, 0 };
429 UINT numFormats = 0;
430 int pf = 0;
431 if (!wglChoosePixelFormatARBFunc(hDC, iAttributes, fAttributes, 1, &pf, &numFormats))
432 {
433 OnError("wglChoosePixelFormatARBFunc failed.");
434 return NULL;
435 }
436
437 PIXELFORMATDESCRIPTOR pfd = {};
438 if (!SetPixelFormat(hDC, pf, &pfd))
439 {
440 OnError("SetPixelFormat failed.");
441 return NULL;
442 }
443
444#ifdef _DEBUG
445 bool UseDebugContext = true;
446#else
447 bool UseDebugContext = false;
448#endif
449
450 // Crete context attributes.
451 GLint attribs[16] = {};
452 {
453 int attribCount = 0;
454 if (UseDebugContext)
455 {
456 attribs[attribCount++] = WGL_CONTEXT_FLAGS_ARB;
457 attribs[attribCount++] = WGL_CONTEXT_DEBUG_BIT_ARB;
458 }
459
460 attribs[attribCount++] = WGL_CONTEXT_MAJOR_VERSION_ARB;
461 attribs[attribCount++] = 3;
462
463 attribs[attribCount++] = WGL_CONTEXT_MINOR_VERSION_ARB;
464 attribs[attribCount++] = 0;
465
466 attribs[attribCount++] = GL_CONTEXT_PROFILE_MASK;
467 attribs[attribCount++] = GL_CONTEXT_CORE_PROFILE_BIT;
468
469 attribs[attribCount] = 0;
470 }
471
472 context = wglCreateContextAttribsARBFunc(hDC, 0, attribs);
473 if (!wglMakeCurrent(hDC, context))
474 {
475 OnError("wglMakeCurrent failed.");
476 return NULL;
477 }
478
479 //
480 if (!sogl_loadOpenGL())
481 {
482 OnError("Failed to load OpenGL.");
483 return NULL;
484 }
485 myGLLoadFunc(wglSwapIntervalEXT, PFNWGLSWAPINTERVALEXTPROC);
486 if (wglSwapIntervalEXT == nullptr)
487 {
488 OnError("wglSwapIntervalEXT not available.");
489 return NULL;
490 }
491
492 // Enable sRGB in the framebuffer if needed.
493 if (g_colorSpace == eColorSpace::sRGBHardware)
494 glEnable(GL_FRAMEBUFFER_SRGB);
495
496 // Basic OpenGL setup.
497 glEnable(GL_DEPTH_TEST);
498 glDepthFunc(GL_LESS);
499 glEnable(GL_CULL_FACE);
500
501 return context;
502}

504bool CreateSRContext(double maxTime)
505{
507
508 const double startTime = (double)GetTickCount64() / 1000.0;
509
510 // Create SR context.
511 while (g_srContext == nullptr)
512 {
513 try
514 {
515 g_srContext = SR::SRContext::create();
516 break;
517 }
518 catch (SR::ServerNotAvailableException e)
519 {
520 // Ignore errors because SR may be starting-up.
521 }
522
523 std::cout << "Waiting for context" << std::endl;
524
525 // Wait a bit.
526 Sleep(100);
527
528 // Abort if we exceed the maximum allowed time.
529 double curTime = (double)GetTickCount64() / 1000.0;
530 if ((curTime - startTime) > maxTime)
531 break;
532 }
533
534 // Wait for display to be ready.
535 bool displayReady = false;
536 while (g_srContext && !displayReady)
537 {
538 // Attempt to get display.
539 SR::Display* display = SR::Display::create(*g_srContext);
540 if (display != nullptr)
541 {
542 // Get the display location, and when it is valid, the device is ready.
543 SR_recti displayLocation = display->getLocation();
544 int64_t width = displayLocation.right - displayLocation.left;
545 int64_t height = displayLocation.bottom - displayLocation.top;
546 if ((width != 0) && (height != 0))
547 {
548 displayReady = true;
549 break;
550 }
551 }
552
553 std::cout << "Waiting for display" << std::endl;
554
555 // Wait a bit.
556 Sleep(100);
557
558 // Abort if we exceed the maximum allowed time.
559 double curTime = (double)GetTickCount64() / 1000.0;
560 if ((curTime - startTime) > maxTime)
561 break;
562 }
563
565
566 // Return if we have a valid context and device is ready.
567 return (g_srContext != nullptr) && displayReady;
568}

597 g_srSystemEventListener = new SRSystemEventListener();
598 g_srSystemEventListener->stream.set(systemSense->openSystemEventStream(g_srSystemEventListener));
599

602 g_viewTextureWidth = display->getRecommendedViewsTextureWidth();
603 g_viewTextureHeight = display->getRecommendedViewsTextureHeight();
604
606

608#ifdef USE_DEPRECATED_WEAVER

610 if (!g_useDeprecatedFramebuffer || (g_demoMode == eDemoMode::StereoImage))
611 g_srWeaver->setInputFrameBuffer(g_viewFramebuffer, g_viewTexture);
612#else

614 if (createWeaverResult != WeaverErrorCode::WeaverSuccess)
615 {
616 OnError("Failed to create weaver");
617 return false;
618 }
619 g_srWeaver->setInputViewTexture(g_viewTexture, g_viewTextureWidth, g_viewTextureHeight, GetTextureFormat());
620#endif
621
623
624 // Set in-shader sRGB conversion if necessary.
625 if (g_colorSpace == eColorSpace::sRGBShader)
626 g_srWeaver->setShaderSRGBConversion(true, true);
627
628 // Initialize context after creating weaver.
629 g_srContext->initialize();
630
631 return true;
632}

944mat4f CalculateModelViewProjectionMatrix(vec3f eye, float elapsedTime)
945{
946 // Compute model matrix.
947 mat4f model = {};
948 {
949 const float size = 60.0; // Size of the object in mm
950 float angle = elapsedTime * 0.75f; // Rotation amount
951
952 const mat4f translation = mat4f::translation(0.0f, 0.0f, 0.0f);
953 const mat4f scaling = mat4f::scaling(size, size, size);
954 const mat4f rotation = mat4f::rotationY(angle);
955
956 model = scaling * translation * rotation;
957 }
958
959 // Compute view matrix.
960 mat4f view = mat4f::identity();
961
962 // Compute projection matrix.
963 mat4f projection = {};
964 {
965 // Implementation of: Kooima, Robert. "Generalized perspective projection." J. Sch. Electron. Eng. Comput. Sci (2009).
966 const float znear = 0.1f;
967 const float zfar = 10000.0f;
968
969 assert((g_ScreenWidthInMM != 0) && (g_ScreenHeightInMM != 0));
970
971 vec3f pa = vec3f(-g_ScreenWidthInMM / 2.0f, g_ScreenHeightInMM / 2.0f, 0.0f);
972 vec3f pb = vec3f( g_ScreenWidthInMM / 2.0f, g_ScreenHeightInMM / 2.0f, 0.0f);
973 vec3f pc = vec3f(-g_ScreenWidthInMM / 2.0f, -g_ScreenHeightInMM / 2.0f, 0.0f);
974
975 vec3f vr = vec3f(1.0f, 0.0f, 0.0f);
976 vec3f vu = vec3f(0.0f, 1.0f, 0.0f);
977 vec3f vn = vec3f(0.0f, 0.0f, 1.0f);
978
979 // Compute the screen corner vectors.
980 vec3f va = pa - eye;
981 vec3f vb = pb - eye;
982 vec3f vc = pc - eye;
983
984 // Find the distance from the eye to screen plane.
985 float distance = -vec3f::dot(va, vn);
986
987 // Find the extent of the perpendicular projection.
988 float l = vec3f::dot(vr, va) * znear / distance;
989 float r = vec3f::dot(vr, vb) * znear / distance;
990 float b = vec3f::dot(vu, vc) * znear / distance;
991 float t = vec3f::dot(vu, va) * znear / distance;
992
993 // Load the perpendicular projection
994 mat4f frustum = mat4f::perspective(l, r, b, t, znear, zfar);
995
996 // Move the apex of the frustum to the origin.
997 mat4f translate = mat4f::translation(-eye);
998
999 // Combine
1000 projection = frustum * translate;
1001 }
1002
1003 // Compute combined matrix.
1004 mat4f mvp = projection * view * model;
1005
1006 return mvp;
1007}

1043 const ULONGLONG curTick = GetTickCount64();
1044 static ULONGLONG prevTick = curTick;

1056 // View texture already contains image. Nothing to do.

1062#if !defined(USE_DEPRECATED_WEAVER)

1065#else
1066 leftEye = g_srEyePairListener->left;
1067 rightEye = g_srEyePairListener->right;

1069
1071
1072 // Render cube into stereo view texture.
1073
1075

1077#if defined(USE_DEPRECATED_WEAVER)

1080 // Use the app-provided view texture.

1117 }
1118
1120 }
1121
1123
1124 // Set viewport to cover entire window.
1125 glBindFramebuffer(GL_FRAMEBUFFER, 0);
1126 glViewport(0, 0, g_windowWidth, g_windowHeight);

1129#if !defined(USE_DEPRECATED_WEAVER)
1130 g_srWeaver->weave();

1154LRESULT CALLBACK WndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam)
1155{
1156 static bool isMoving = false;
1157 switch (message)
1158 {
1159
1160 case WM_CLOSE:
1161 DestroyWindow(hWnd);
1162 return 0; // Message handled
1163
1164 case WM_DESTROY:
1165 PostQuitMessage(0); // Causes GetMessage() to return 0 and end message loop
1166 return 0; // Message handled
1167
1168 // Handle keypresses.
1169 case WM_KEYDOWN:
1170 switch (wParam)
1171 {
1172 // Quit.
1173 case VK_ESCAPE:
1174 PostMessage(hWnd, WM_CLOSE, 0, 0); // Gracefully initiate close
1175 return 0; // Message handled
1176
1177 // Toggle windowed/fullscreen.
1178 case VK_F11:
1179 g_fullscreen = !g_fullscreen;
1180 SetFullscreen(hWnd, g_fullscreen);
1181 return 0; // Message handled
1182
1183 // Simulate destruction of context.
1184 case 'c':
1185 case 'C':
1186 {
1187 SR::SystemEvent newSystemEvent{
1188 (uint64_t)std::chrono::system_clock::now().time_since_epoch().count(),
1189 SR_eventType::ContextInvalid,
1190 "ContextInvalidEvent",
1191 };
1192 g_srSystemEventListener->accept(newSystemEvent);
1193 return 0; // Message handled
1194 }
1195 }
1196 break;
1197
1198 // Limit minimum size of window to not cause crash when width or height becomes 0
1199 case WM_GETMINMAXINFO:
1200 {
1201 MINMAXINFO* mmi = reinterpret_cast<MINMAXINFO*>(lParam);
1202
1203 // Set the minimum tracking size (in pixels)
1204 mmi->ptMinTrackSize.x = 100; // Min width
1205 mmi->ptMinTrackSize.y = 100; // Min height
1206
1207 return 0; // Message handled
1208 }
1209
1210 // Keep track of window size. Assumption: rendering is not in a separate thread (needs mutex)
1211 case WM_SIZE:
1212 if (wParam != SIZE_MINIMIZED &&
1213 (
1214 g_windowWidth != LOWORD(lParam) ||
1215 g_windowHeight != HIWORD(lParam)
1216 ) )
1217 {
1218 g_windowWidth = LOWORD(lParam);
1219 g_windowHeight = HIWORD(lParam);
1220 if (g_sceneReady) Render(true);
1221 }
1222 break;
1223
1224 case WM_ENTERSIZEMOVE:
1225 // Assumption: rendering is not in a separate thread
1226 // When user holds an edge/titlebar, but not moves, WM_PAINT will never be called. Invalidate the rect here, and in WM_PAINT invalididate the window.
1227 // We need to rerender in order to keep 3D directed to the user's position.
1228 isMoving = true;
1229 InvalidateRect(hWnd, nullptr, false);
1230 break;
1231
1232 case WM_EXITSIZEMOVE:
1233 isMoving = false;
1234 break;
1235
1236 case WM_PAINT:
1237 // Don't waste cycles on the default paint handler.
1238 if (isMoving)
1239 {
1240 // Assumption: rendering is not in a separate thread
1241 // Keep on rendering while windows is holding WndProc hostage when user drags titlebar or resizes the window.
1242 Render();
1243 // Do not validate the window here, so WM_PAINT will be sent again
1244 }
1245 else
1246 {
1247 // Validate the window, so no more paint messages are sent (rendering will be done in the main loop)
1248 PAINTSTRUCT ps;
1249 if (BeginPaint(hWnd, &ps))
1250 EndPaint(hWnd, &ps);
1251 }
1252 return 0; // return value seems not to matter. Microsoft uses 0 in the examples.
1253
1254 case WM_ERASEBKGND:
1255 // We fully draw the entire surface already, no need to erase the background. It also prevents showing a white backround when we dont render fast enough.
1256 return 1; // Message handled
1257 }
1258
1259 return DefWindowProc(hWnd, message, wParam, lParam);
1260}
